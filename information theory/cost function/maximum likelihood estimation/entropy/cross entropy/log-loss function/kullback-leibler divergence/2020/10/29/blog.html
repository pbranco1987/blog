Cross Entropy and Maximum Likelihood Estimation

In this article, I will discuss what Cross Entropy means from an information theoretical point of view and how it relates to key concepts in Machine Learning, including the log-loss cost function in linear regression classification and in parameter estimation.

import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd
import string as string

def entropy(p,eps):
    p = p+eps
    return -np.dot(p,np.log2(p))

def cross_entropy(p,q,eps):
    q = q+eps
    return -np.dot(p,np.log2(q))

def KL(p,q,eps):
    p = p+eps; q = q+eps
    return -np.dot(p,np.log2(p/q))

eps = 1e-15

nel = 100
p_list = np.linspace(0,1,nel+1)
p_set = [np.array([p, 1-p]) for p in p_list]
Hp = [entropy(value,eps) for value in p_set]
dic = {'P(x=1)':p_list, 'Hp':Hp}

dfentropy = pd.DataFrame(dic)
sns.set_style("whitegrid")
fig_dims = (10, 6)
fig,ax = plt.subplots(figsize=fig_dims)
ax.set(xlim=(0,1), ylim=(0,1), xticks=np.linspace(0,1,11))
sns.lineplot(x="P(x=1)", y="Hp", ax=ax, data=dfentropy)
plt.title('Binary Entropy')
